교사용 웹서비스 수집 설정

Google Apps Script 웹앱 URL을 아래 칸에 붙여 넣으면 학생 제출 결과가 Google Sheet로 전송됩니다. URL은 이 브라우저에만 저장됩니다.

Google Apps Script 코드 보기
function doPost(e) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('응답') || SpreadsheetApp.getActiveSpreadsheet().insertSheet('응답');
  const data = JSON.parse(e.postData.contents);
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(['제출시간','이름','학년반','캐릭터','점수','총점','정답률','느낀점','어려웠던점','다음에 만들고 싶은 것','문항별응답JSON']);
  }
  sheet.appendRow([new Date(), data.name, data.className, data.avatar, data.score, data.total, data.percent, data.feeling, data.difficulty, data.nextProject, JSON.stringify(data.answers)]);
  return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
}
